-- d. Create a function GetBestFilm
RETURNS BIGINT
AS
BEGIN
    DECLARE @BestFilmID BIGINT

    SELECT @BestFilmID = FID
    FROM Films
    WHERE PROD_YEAR = @P_YEAR AND RATING = (
        SELECT MAX(RATING)
        FROM Films
        WHERE PROD_YEAR = @P_YEAR
    )

    RETURN @BestFilmID
END
